    <!-- Table of root sessions, sudo, and su events with severity -->
    <panel>
      <title>Shell Execution &amp; Root Terminal Detection (10 min)</title>
      <table>
        <search>
          <query>
            source="/var/log/auth.log" ("session opened for user root" OR "sudo:" OR "su:" OR "COMMAND=")
            | rex "for user (?<target_user>\w+)"
            | rex "by (?<initiated_by>\w+)"
            | rex "from (?<ip>\d+\.\d+\.\d+\.\d+)"
            | rex "COMMAND=(?<command>.+)"
            | eval event_type=case(
                match(_raw,"COMMAND="),                     "SUDO COMMAND",
                match(_raw,"session opened for user root"), "ROOT SESSION OPENED",
                match(_raw,"su:"),                          "SU ATTEMPT",
                true(),                                     "SUSPICIOUS AUTH"
              )
            | eval severity=case(
                event_type="ROOT SESSION OPENED", "CRITICAL",
                event_type="SUDO COMMAND",        "HIGH",
                event_type="SU ATTEMPT",          "MEDIUM",
                true(),                           "LOW"
              )
            | sort -_time
            | head 20
            | eval Time=strftime(_time,"%Y-%m-%d %H:%M:%S")
            | table Time, severity, event_type, target_user, initiated_by, ip, command
            | rename severity as Severity, event_type as "Event Type",
                     target_user as "Target User", initiated_by as "Initiated By",
                     ip as "Source IP", command as Command
          </query>
          <earliest>-30m</earliest>
          <latest>now</latest>
          <refresh>15s</refresh>
          <refreshType>delay</refreshType>
        </search>
        <option name="drilldown">row</option>
